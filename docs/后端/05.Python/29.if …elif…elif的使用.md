# if …elif…elif的使用

```python
nub = int(input())
# #多个if语句，语句和语句之间，不存在关联，各走各的
# if 1 < nub < 3:
#     print('小于3')
#
# if 3 < nub < 7:
#     print('小于7')
#
# if 7 < nub < 9:
#     print('小于9')


if 1 <= nub <= 3:
    print('小于或等于3')
elif 3 <= nub <= 7:         #不满足第一if的条件走一下一直走到满足if条件的语句
    print('小于或等于7')
elif 7 <= nub <= 9:
    print('小于或等于9')
else:print('请输入1-9的数字')  #否则语句，判断不在if条件中的然后出此结果


#if语句的嵌套
#在python中，使用强制缩进来表示语句之间的结构
a =input('是否同意请输入Y/N:')
if a == 'Y':
    print('你已同意')
    next = input('下一步Y/N')
    if next == 'Y':
        print('恭喜你')
    else:
        print('你已取消')
else:
    print('你已经取消同意')
```
